-- FileSystem API for MilkOS Core by Redall

function listFiles(path)
 local directoryFiles = fs.list(path); 
 for _, fileDir in ipairs(directoryFiles) do 
   if fs.isDir(fileDir) then
    print(fileDir .. " [DIR]");
   else
    if fs.isReadOnly(fileDir) then
	 print(fileDir .. " [LKD]");
	else
	 print(fileDir);
	end
   end
 end 
end
